K_numSeqs <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = numSeqs)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against numSeqs")
K_n50 <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = n50)) + geom_boxplot() + 
    ggtitle("Plot of K parameter against n50")
K_smallest <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = smallest)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against smallest")
K_largest <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = largest)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against largest")
K_totLen <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = totLen)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against totLen")
K_time <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = time)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against time")
K_gz_size <- ggplot(data = merge_leaff_soapdt, aes(x = factor(K), y = gz_size)) + 
    geom_boxplot() + ggtitle("Plot of K parameter against gz_size")
grid.arrange(K_numSeqs, K_n50, K_smallest, K_largest, K_totLen, K_time, K_gz_size, 
    ncol = 2, nrow = 4)
